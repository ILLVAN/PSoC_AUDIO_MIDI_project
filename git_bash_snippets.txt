USEFUL GIT BASH SNIPPETS

// UPLOAD from scratch:
navigate to folder cd "xxx"/"yyy" "						// wherever the local repo should be located
git init
git status
git add --all
git status
git commit -m "Initial commit"                                    // any message
git remote add origin https://github.com/NAME/repo_name.git		// change user & repo name	  
git branch -M main
git push -u origin main


// RESET COMMIT
git rm --cached <file>								// insert file name


// FETCH & MERGE from remote to local
navigate to folder cd "xxx"/"yyy" "						// wherever the local repo is located
git remote -v									// check if repo is connected to remote
git remote add origin https://github.com/NAME/repo_name.git		// change user & repo name, only needed if not connected
git pull --verbose								// fetch & merge with outputs


// SHOW ALL BRANCHES
git log --all             // without --all, only historically lower branches are shown

// SHOW BRANCH POINTERS
git log --oneline --decorate

// EXIT BRANCH
git checkout

// SWITCH TO OTHER BRANCH
git checkout otherbranch
